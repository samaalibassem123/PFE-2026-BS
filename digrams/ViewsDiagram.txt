    BIOTIME_EMPLOYEE_VIEW {
        int ep_emp_id PK
        varchar emp_fullname
        varchar emp_code
        varchar email
        int department_id
        date hire_date
    }

    BIOTIME_ATT_LEAVE_VIEW {
        datetime start_time
        datetime end_time
        datetime apply_time
        float leave_day
        varchar apply_reason
        varchar name
        int emp_id FK
        varchar approver_instance
        datetime approval_time
        varchar email
        varchar first_name
    }

    CHECK_IN_OUT_VIEW {
        int id PK
        datetime clock_in
        datetime clock_out
        date att_date
        int emp_id FK
        varchar email
        varchar first_name
        int department_id
    }

    %% ------------------ EASY PROJECT ------------------
    EASY_EMPLOYEE_VIEW {
        int ep_emp_id PK
        varchar emp_fullname
        varchar email
    }

    EASY_ATTENDANCE_FULL_VIEW {
        int id PK
        datetime created_at
        datetime updated_at
        datetime arrival
        datetime departure
        float hours
        varchar arrival_user_ip
        varchar departure_user_ip
        int ep_emp_id FK
        varchar emp_fullname
        varchar email
        int activity_id FK
        varchar activity_name
    }
