        int department_id

        datetime start_time
        datetime end_time
        datetime apply_time
        float leave_day
        varchar apply_reason
        varchar name
        int pay_code_id
        int emp_id FK
        varchar approver_instance
        datetime approval_time
        varchar email
        varchar first_name
    }

        int id PK
        datetime clock_in
        datetime clock_out
        date att_date
        int emp_id FK
        varchar email
        varchar first_name
        int department_id
    }
